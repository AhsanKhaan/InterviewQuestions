function LongestWord(sen) { 

  let cleanArray = sen.replace(/[^a-zA-Z ]/g,"").split(" ");
  let maxWord = '';
  cleanArray.forEach((element)=>{
    if(element.length > maxWord.length){
      maxWord = element;
    }
  })
  // code goes here  
  return maxWord; 

}
